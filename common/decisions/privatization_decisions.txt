	privatize_high_value_government_land = {
		allowed = { always = yes }
		icon = GFX_decision_generic_forestry
		days_re_enable = 270
		cost = 150
		available = {
			any_owned_state = {
				is_controlled_by = ROOT
				OR = {
					has_state_category = megalopolis
					has_state_category = metropolis
					has_state_category = large_city
				}
				OR = {
					free_building_slots = {
					    building = arms_factory
					    size > 1
					    include_locked = no
					}
					free_building_slots = {
					    building = industrial_complex
					    size > 1
					    include_locked = no
					}
				}
			}
			if = {
				limit = {
					has_government = democratic
				}
				democratic > 0.61
			}
			if = {
				limit = {
					has_government = neutrality
				}
				neutrality > 0.61
			}
			if = {
				limit = {
					has_government = communism
				}
				communism > 0.61
			}
			if = {
				limit = {
					has_government = fascism
				}
				fascism > 0.61
			}
		}
		complete_effect = {
			add_stability = 0.15
			random_owned_controlled_state = {
				limit = {
					OR = {
						has_state_category = megalopolis
						has_state_category = metropolis
						has_state_category = large_city
					}
					OR = {
						free_building_slots = {
						    building = arms_factory
						    size > 1
						    include_locked = no
						}
						free_building_slots = {
						    building = industrial_complex
						    size > 1
						    include_locked = no
						}
					}
				}
				add_extra_state_shared_building_slots = -1
			}
			if = {
				limit = {
					has_government = democratic
				}
				add_popularity = {
				    ideology = democratic
				    popularity = -0.06
				}
			}
			if = {
				limit = {
					has_government = neutrality
				}
				add_popularity = {
				    ideology = neutrality
				    popularity = -0.06
				}
			}
			if = {
				limit = {
					has_government = communism

				}
			}
